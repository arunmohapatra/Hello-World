package com.example.kubeproxy;

import org.springframework.boot.SpringApplication;
import org.springframework.boot.autoconfigure.SpringBootApplication;
import org.springframework.http.HttpHeaders;
import org.springframework.http.HttpMethod;
import org.springframework.http.server.reactive.ServerHttpRequest;
import org.springframework.http.server.reactive.ServerHttpResponse;
import org.springframework.web.bind.annotation.RequestMapping;
import org.springframework.web.bind.annotation.RestController;
import org.springframework.web.reactive.function.client.WebClient;

import reactor.core.publisher.Mono;

@SpringBootApplication
@RestController
public class KubernetesApiProxyApplication {

    public static void main(String[] args) {
        SpringApplication.run(KubernetesApiProxyApplication.class, args);
    }

    // Kubernetes API server
    private static final String KUBE_API_SERVER = "https://kube-apiserver:6443";

    private final WebClient webClient = WebClient.builder().build();

    /**
     * Catch all incoming kubectl requests
     */
    @RequestMapping("/**")
    public Mono<Void> proxy(ServerHttpRequest request, ServerHttpResponse response) {

        String path = request.getURI().getRawPath();
        String query = request.getURI().getRawQuery();

        String targetUrl = KUBE_API_SERVER + path;

        if (query != null) {
            targetUrl += "?" + query;
        }

        HttpMethod method = request.getMethod();

        // Copy headers
        HttpHeaders headers = new HttpHeaders();
        headers.addAll(request.getHeaders());

        return webClient
                .method(method)
                .uri(targetUrl)
                .headers(h -> h.addAll(headers))
                .body(request.getBody(), byte[].class)
                .exchangeToMono(clientResponse -> {

                    response.setStatusCode(clientResponse.statusCode());

                    response.getHeaders().addAll(clientResponse.headers().asHttpHeaders());

                    return response.writeWith(clientResponse.bodyToFlux(org.springframework.core.io.buffer.DataBuffer.class));
                });
    }
}
